
-- When set identity for table's column on oracle database it will create the trigger for that table
and use the sequence to insert the new value for that column

CREATE OR REPLACE EDITIONABLE TRIGGER "SBAPI"."SBX_RPT_ENG_RENDER_DETAILS_TRG"
BEFORE INSERT ON SBX_RPT_ENG_RENDER_DETAILS
FOR EACH ROW
BEGIN
<<COLUMN_SEQUENCES>>
BEGIN
  IF INSERTING AND :NEW.REPORT_REQUEST_ID IS NULL THEN
    SELECT SBX_RPT_ENG_RENDER_DETAILS_SEQ.NEXTVAL INTO :NEW.REPORT_REQUEST_ID FROM SYS.DUAL;
  END IF;
END COLUMN_SEQUENCES;
END;
