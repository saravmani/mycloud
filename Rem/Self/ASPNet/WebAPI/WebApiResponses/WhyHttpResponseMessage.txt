public HttpResponseMessage GetEmployee(int id)
{
   try
   {
      Employee emp = EmployeeContext.Employees.Where(e => e.Id == id).FirstOrDefault();

      if (emp != null)
      {
         return Request.CreateResponse<Employee>(HttpStatusCode.OK, emp);
      }
      else
      {
         return Request.CreateErrorResponse(HttpStatusCode.NotFound, " Employee Not Found");
      }
   }
   catch (Exception ex)
   {
      // Log exception code goes here
      return Request.CreateErrorResponse(HttpStatusCode.InternalServerError, "Error occured while executing GetEmployee‚Äù);
   }
}
