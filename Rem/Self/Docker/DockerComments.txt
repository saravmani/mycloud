docker images   //to list images
docker pull image name

docker rmi <image id> // to remove image
docker rm <container id> //to remove container

docker container list // lists the containers with running status
docker container list -a //lists the containers with or without running status


docker run -p 1000:1000 <image name>
docker ps //show running list of running images


docker log <container id>

docker stop $(docker ps -aq) //to stop all containers
docker rmi $(docker images -q) //to remove all images

To run the conatainer interactive mode
-------------------------------------------
docker run -it ngnix:1.13.8 /bin/bash
//this command runs bash in docker
//two switches in this command
i - to run container in interactive mode
t - commands typed in our terminal, will get executed within container's terminal
    and produces output in outside
